\section{API Endpoint Documentation}
The EduMeter REST API uses standard HTTP methods to perform CRUD operations. Access levels are strictly enforced via the \texttt{@AdminGuard} and \texttt{@AuthGuard} filters. Information sending and retrieval is done in JSON format. 

\subsection{Authentication (\texttt{/auth})}
\begin{description}
	\item[\POST \quad \texttt{/auth/sendPin}] \hfill \\
	\textbf{Access:} Public \\
	\textbf{Query Params:} \texttt{admin} (boolean) \\
	\textbf{Body:} \texttt{\{ "email": "string" \}} \\
	\textbf{Response:} \texttt{\{ "challengeId": 1234 \}} \\
	\textbf{Description:} Initiates the login process by accepting an email and an \texttt{admin} boolean. It validates the email domain (enforcing \texttt{@edu.unifi.it} for students) and generates a 4-digit PIN challenge. Returns a \texttt{challengeId} for the next step. \\
	\textbf{Test Reference:} \ref{test:testSendPin_MissingEmail:Auth}, \ref{test:testSendPin_Admin_Valid:Auth}, \ref{test:testSendPin_Student_ValidDomain:Auth}, \ref{test:testSendPin_Student_InvalidDomain:Auth}
	
	\item[\POST \quad \texttt{/auth/login}] \hfill \\
	\textbf{Access:} Public \\
	\textbf{Query Params:} \texttt{admin} (boolean) \\
	\textbf{Body:} \texttt{\{ "challengeId": 1234, "pin": "string" \}} \\
	\textbf{Response:} \texttt{\{ "token": "JWT\_ACCESS\_TOKEN" \}} \\
	\textbf{Description:} Completes the identity challenge. Validates the provided \texttt{pin} against the \texttt{challengeId}. If successful, it returns a JWT session token for further requests.\\
	\textbf{Test Reference:} \ref{test:testLogin_Vaid:Auth}, \ref{test:testLogin_ExpiredPin:Auth}, \ref{test:testLogin_WrongPin:Auth}, \ref{test:testLogin_MissingFields:Auth}, \ref{test:testLogin_NotFound:Auth}
	
	\item[\POST \quad \texttt{/auth/logout}] \hfill \\
	\textbf{Access:} \texttt{@AuthGuard} \\
	\textbf{Header Params:} \texttt{bearer} (String) \\
	\textbf{Response:} \texttt{\{ "message": "Token revoked" \}} \\
	\textbf{Description:} Invalidates the provided Bearer token by adding it to the server-side revocation list, effectively ending the user session.\\
	\textbf{Test Reference:} \ref{test:testLogout:Auth}
	
\end{description}

	\textbf{Test Reference:} \ref{test:testSearchSchool_All:School}, \ref{test:testSearchSchool_Filtered:School}, \ref{test:testSearchSchool_Empty:School}
	
	\item[\POST \quad \texttt{/school}] \hfill \\

	\textbf{Response:} \texttt{\{ "id": int, "message": "School created" \}} \\
	\textbf{Description:} Creates a new School entity. Requires a JSON body containing the school name.\\
	\textbf{Test Reference:} \ref{test:testCreateSchool:School}
	
	\item[\PUT \quad \texttt{/school}] \hfill \\
	\textbf{Access:} \texttt{@AdminGuard} \\
	\textbf{Body:} \texttt{\{ "id": int, "name": "string" \}} \\
	\textbf{Response:} \texttt{\{ "message": "School updated" \}} \\
	\textbf{Description:} Updates an existing school's details. The school must exist; otherwise, it returns a 404 error.\\
	\textbf{Test Reference:} \ref{test:testUpdateSchool_Found:School}, \ref{test:testUpdateSchool_NotFound:School}, \ref{test:testUpdateSchool_Invalid:School}
	
	\item[\GET \quad \texttt{/school/\{school\_id\}}] \hfill \\
	\textbf{Access:} Public \\
	\textbf{Path Params:} \texttt{school\_id} (int) \\
	\textbf{Response:} \texttt{\{ "id": int, "name": "string" \}} \\
	\textbf{Description:} Returns the information of the school identified by the given school Id.\\
	\textbf{Test Reference:} \ref{test:testGetSchool_Found:School}, \ref{test:testGetSchool_NotFound:School}
	
	\item[\DELETE \quad \texttt{/school/\{school\_id\}}] \hfill \\
	\textbf{Access:} \texttt{@AdminGuard} \\
	\textbf{Path Params:} \texttt{school\_id} (int) \\
	\textbf{Response:} \texttt{\{ "message": "School deleted" \}} \\
	\textbf{Description:} Deletes a school and its associated data from the persistence layer.\\
	\textbf{Test Reference:} \ref{test:testDeleteSchool_Found:School}, \ref{test:testDeleteSchool_NotFound:School}
\end{description}

\subsection{Review Management (\texttt{/review})}
\begin{description}
	\item[\GET \quad \texttt{/review}] \hfill \\
	\textbf{Access:} Public / Mixed \\
	\textbf{Query Params:} \texttt{school\_id} (int), \texttt{degree\_id} (int), \texttt{course\_id} (int), \texttt{professor\_id} (int) \\
	\textbf{Response:} \texttt{\seqsplit{List<\{ "id": int, "comment": "string", "date": "string", "enjoyment": int, "difficulty": int, "teachingId": int, "upvotes": int, "upvotedByUser": boolean \}>}} \\
	\textbf{Description:} Searches for published reviews based on academic filters. If the request is authenticated, the \texttt{upvotedByUser} field is populated based on the requester's identity.\\
	\textbf{Test Reference:} \ref{test:testSearch_Auth:PubReview}, \ref{test:testSearch_Unauth:PubReview}
	
	\item[\POST \quad \texttt{/review}] \hfill \\
	\textbf{Access:} \texttt{@AuthGuard} \\
	\textbf{Body:} \texttt{\seqsplit{\{ "comment": "string", "enjoyment": int, "difficulty": int, "teachingId": int \}}} \\
	\textbf{Response:} \texttt{\{ "id": int, "message": "string" \}} \\
	\textbf{Description:} Submits a new review directly. The system automatically sets the current date, initializes upvotes to zero, and extracts the creator's hash from the security context.\\
	\textbf{Test Reference:} \ref{test:testPublish:PubReview}
	
	\item[\PUT \quad \texttt{/review}] \hfill \\
	\textbf{Access:} \texttt{@AdminGuard} \\
	\textbf{Body:} \texttt{\{ "id": int, "comment": "string", "enjoyment": int, "difficulty": int, "teachingId": int \}} \\
	\textbf{Response:} \texttt{\{ "message": "Review updated" \}} \\

	\textbf{Test Reference:} \ref{test:testUpdate:PubReview}
	
	\item[\GET \quad \texttt{/review/\{review\_id\}}] \hfill \\

	\textbf{Description:} Retrieves a specific published review by its ID. It includes the voting status if the user is logged in.\\
	\textbf{Test Reference:} \ref{test:testGet_Authenticated_Found:PubReview}, \ref{test:testGet_Unauthenticated_NotFound:PubReview}
	
	\item[\POST \quad \texttt{/review/\{review\_id\}/vote}] \hfill \\

	\textbf{Description:} Toggles an upvote for the current user on the specified review. The review must exist to complete the operation.\\
	\textbf{Test Reference:} \ref{test:testToggleVote:PubReview}
	
	\item[\DELETE \quad \texttt{/review/\{review\_id\}}] \hfill \\

	\textbf{Path Params:} \texttt{review\_id} (int) \\
	\textbf{Response:} \texttt{\{ "message": "Review deleted" \}} \\
	\textbf{Description:} Permanently removes a published review. This endpoint verifies existence before performing the deletion.\\
	\textbf{Test Reference:} \ref{test:testDelete:PubReview}
